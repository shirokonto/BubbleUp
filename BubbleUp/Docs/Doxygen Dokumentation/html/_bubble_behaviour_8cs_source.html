58 /*
59 * Detects different GameObjects on collision with bubble

61 private void OnCollisionEnter(Collision other)
62 {
63 switch (other.gameObject.tag)
64 {
65 case "InfoObject":
66 HitInfoItem(other.gameObject);
67 break;
68 case "MinimizeBubble":
69 HitMinimizeBubble(other.gameObject);

71 case "Virus":

80 /*
81 * Points are collected and the localPoints are set if the correct InfoType is collected.
82 * Bubble size changes depending on InfoType
83 * Bubble bursts if it got hit by wrong InfoTypes three times
84 */
85 private void HitInfoItem(GameObject infoItem)
86 {
87 if (!infoItem.transform.name.Contains(correctInfoType.Get()))
88 {
89 _hit += 1;
90 _localPoints = (_localPoints -= MINUS_POINT) <= 0 ? 0 : (_localPoints);
91 points.Set(_localPoints);
92 transform.localScale += _scaleChange;
93 }
94 else
95 {
96 _localPoints += PLUS_POINTS;
97 points.Set(_localPoints);
98 }
99
100 if (_hit == 3)
101 {
102 AudioSource.PlayClipAtPoint(bubblePopSound, transform.position);
103 bubbleIsPopped.Set(true);
104 Destroy(this.gameObject);
105 bubblePop.Play();
106 Menu.IsGameOver = true;
107 onLoadGameOver.Raise();
108 }
109
110 infoItem.GetComponent<ItemMover>().ResetPositionAndRotationBeforeRespawn();
111 }

116 private void HitMinimizeBubble(GameObject minimizeItem)
117 {
118 if (_hit >= 1 && _hit < 3)

120 _hit -= 1;
121 transform.localScale -= _scaleChange;
122 }
123 minimizeItem.GetComponent<ItemMover>().ResetPositionAndRotationBeforeRespawn();
124 StartCoroutine(ShowIconWhenMinimizeHit());
125 }
